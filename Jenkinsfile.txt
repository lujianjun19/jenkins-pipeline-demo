pipeline {
   agent any
   stages {
      stage('Source') {
        steps {
           git branch: 'master', url: 'https://github.com/lujianjun19/jenkins-pipeline-demo.git'
        }
    }
      stage('Build') {
        steps {
           script {
              docker.build("webapp:${env.BUILD_ID}", "./WebApp")
           }
        }
      }
   }
}
